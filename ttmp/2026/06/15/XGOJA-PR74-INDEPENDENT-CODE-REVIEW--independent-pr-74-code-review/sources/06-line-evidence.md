   206	func resolveValueSource(req *RequestDTO, source ValueSource) (string, error) {
   207		switch source.Kind {
   208		case ValueSourceParam:
   209			value, ok := req.Params[source.Key]
   210			if !ok || value == "" {
   211				return "", fmt.Errorf("missing route parameter %q", source.Key)
   212			}
   213			return value, nil
   214		case ValueSourceQuery:
   215			value, ok := req.Query[source.Key]
   216			if !ok {
   217				return "", fmt.Errorf("missing query value %q", source.Key)
   218			}
   219			return stringifySourceValue(value, source.Key)
   220		case ValueSourceBody:
   221			body, ok := req.Body.(map[string]any)
   222			if !ok {
   223				return "", fmt.Errorf("body source %q requires object body", source.Key)
   224			}
   225			value, ok := body[source.Key]
   226			if !ok {
   227				return "", fmt.Errorf("missing body value %q", source.Key)
   228			}

   234		}

   525	Go-backed modules can also expose mountable HTTP handlers. Express can mount
   526	those handlers while JavaScript remains the composition layer:
   527	
   528	```ts
   529	import express from "express"
   530	import sessionstream from "sessionstream"
   531	
   532	__package__({ name: "sites" })

   536	  const hub = sessionstream.hub({ schemas })

   539	  app.mount("/ws", sessionstream.webSocket.server(hub))
   540	}

   542	
   543	The mounted object must carry the shared `gojahttp` hidden `http.Handler` ref.
   544	This is how a Go-backed transport such as a WebSocket server can be mounted
   545	without reimplementing upgrade handling in JavaScript.
   546	
   547	Express route patterns and mounted handlers use different matching semantics:
   548	
   549	- `app.get("/users/:id", ...)` captures one segment as `req.params.id`.
   550	- `app.get("/assets/*", ...)` matches the rest of the path but does not expose a splat capture today.
   551	- `app.mount("/ws", handler)` uses prefix matching and gives the request to the Go handler.
   552	
   553	`app.mount()` is intentionally not a JavaScript parameter/wildcard router. Mount Go-backed transports at a stable prefix and let the Go handler do any detailed matching by reading `r.URL.Path` or by delegating to its own router.
